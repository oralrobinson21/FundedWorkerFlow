2️⃣ Add a working “Cancel Request” button on job detail

Now fix cancelling.

Find your JobDetailScreen (or similar) and add something like this near the bottom of your component:

// inside JobDetailScreen component

import { Alert, TouchableOpacity, Text, StyleSheet } from 'react-native';
import { useAppContext } from '../context/AppContext';

export const JobDetailScreen = ({ route }) => {
  const { taskId } = route.params;
  const { tasks, user, cancelTask } = useAppContext();

  const task = tasks.find(t => t.id === taskId);

  if (!task) {
    return (
      <View style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
        <Text>Task not found.</Text>
      </View>
    );
  }

  const canCancel =
    (task.status === 'requested' || task.status === 'accepted') &&
    task.posterId === user?.id;

  const handleCancel = () => {
    Alert.alert(
      'Cancel job',
      'Are you sure you want to cancel this job?',
      [
        { text: 'No' },
        {
          text: 'Yes, cancel',
          style: 'destructive',
          onPress: async () => {
            try {
              console.log('Cancelling task', task.id);
              await cancelTask(task.id);
            } catch (err) {
              console.error(err);
              Alert.alert('Error', 'Could not cancel job. Please try again.');
            }
          },
        },
      ]
    );
  };

  return (
    <View style={{ flex: 1 }}>
      {/* your existing job content here */}

      {canCancel && (
        <TouchableOpacity style={cancelStyles.button} onPress={handleCancel}>
          <Text style={cancelStyles.text}>Cancel Request</Text>
        </TouchableOpacity>
      )}
    </View>
  );
};

const cancelStyles = StyleSheet.create({
  button: {
    marginTop: 16,
    marginHorizontal: 16,
    paddingVertical: 12,
    borderRadius: 8,
    backgroundColor: '#b00020',
  },
  text: { textAlign: 'center', color: 'white', fontWeight: '600' },
});

And make sure cancelTask in your AppContext actually calls the backend:

// in AppContext

const cancelTask = async (taskId: string) => {
  await fetch(`${API_BASE_URL}/api/tasks/${taskId}/cancel`, {
    method: 'POST',
    headers: { Authorization: `Bearer ${token}` },
  });

  setState(prev => ({
    ...prev,
    tasks: prev.tasks.map(t =>
      t.id === taskId ? { ...t, status: 'canceled' } : t
    ),
  }));
};

(Use whatever auth/token system you have.)

Reload app:
	•	If you tap the green button and no alert pops, something global is eating events:
